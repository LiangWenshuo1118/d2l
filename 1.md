```
import numpy as np
from mendeleev import element

# Define the list of elements
elements_list = ["Li", "Na", "Mg", "Cl", "K"]

# Create a matrix with specific properties for each element
matrix = []
for e in elements_list:
    elem = element(e)
    properties = [
        elem.atomic_number,
        elem.period,
        elem.en_pauling,
        elem.atomic_volume,
        elem.atomic_weight,
        elem.dipole_polarizability,
        elem.fusion_heat,
        elem.ionenergies.get(1) if 1 in elem.ionenergies else None,
        elem.covalent_radius_bragg
    ]
    matrix.append(properties)

# Convert list to numpy array
matrix = np.array(matrix)
print(matrix)

```
compounds = {
    "LiCl": [1, 0, 0, 1, 0],
    "NaCl": [0, 1, 0, 1, 0],
    "KCl":  [0, 0, 0, 1, 1],
    "MgCl2":[0, 0, 1, 2, 0]
}

system = ["LiCl", "NaCl", "MgCl2", "KCl"]
percentages = [5.7, 0, 94.3, 0]  # Mole percentages for LiCl and MgCl2

# Calculate weighted sums for each element
weighted_sum = [0] * 5
for compound, percent in zip(system, percentages):
    weighted_sum = [sum(x) for x in zip(weighted_sum, [x * percent for x in compounds[compound]])]

# Calculate the percentage of each element
total_moles = sum(weighted_sum)
percentages_of_elements = [x / total_moles for x in weighted_sum]

# Convert percentages to a numpy column vector
percentages_of_elements = np.array(percentages_of_elements).reshape(-1, 1)
print(percentages_of_elements)
